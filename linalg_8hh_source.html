112  bisection(T a, T b, F func, T eps, int max_iter = 30) {
113  T c, fc;
114  int i = 0;
115  do {
116  c = T(0.5) * (a + b);
117  fc = func(c);
118  if (func(a) * fc < T(0)) b = c;
119  if (func(b) * fc < T(0)) a = c;
120  i++;
121  } while (i < max_iter && (b - a) > eps && std::abs(fc) > eps);
122  // TODO eps is used as absolute error and as a minimal interval size
123  return c;
124  }
