150  bisection(T a, T b, F func, T eps, int max_iter = 30) {
151  T c, fc;
152  int i = 0;
153  do {
154  c = T(0.5) * (a + b);
155  fc = func(c);
156  if (func(a) * fc < T(0)) b = c;
157  if (func(b) * fc < T(0)) a = c;
158  i++;
159  } while (i < max_iter && (b - a) > eps && std::abs(fc) > eps);
160  // TODO eps is used as absolute error and as a minimal interval size
161  return c;
162  }
