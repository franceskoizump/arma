187  bisection(T a, T b, F func, T eps, int max_iter = 30) {
188  T c, fc;
189  int i = 0;
190  do {
191  c = T(0.5) * (a + b);
192  fc = func(c);
193  if (func(a) * fc < T(0)) b = c;
194  if (func(b) * fc < T(0)) a = c;
195  i++;
196  } while (i < max_iter && (b - a) > eps && std::abs(fc) > eps);
197  // TODO eps is used as absolute error and as a minimal interval size
