193  template <class T, class F>
194  T
195  bisection(T a, T b, F func, T eps, int max_iter = 30) {
196  T c, fc;
197  int i = 0;
198  do {
199  c = T(0.5) * (a + b);
200  fc = func(c);
201  if (func(a) * fc < T(0)) b = c;
202  if (func(b) * fc < T(0)) a = c;
203  i++;
204  } while (i < max_iter && (b - a) > eps && std::abs(fc) > eps);
205  // TODO eps is used as absolute error and as a minimal interval size
206  return c;
207  }
